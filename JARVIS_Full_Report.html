import os
import numpy as np
import speech_recognition as sr
import whisper

class HybridVoiceEngine:
    def __init__(self, model_size="tiny"):
        self.recognizer = sr.Recognizer()
        
        # Optimize for ambient noise environments
        self.recognizer.dynamic_energy_threshold = True
        self.recognizer.energy_threshold = 400
        self.recognizer.pause_threshold = 0.8
        
        print(f"Loading Whisper '{model_size}' model for Auto-Detect Multilingual ASR...")
        try:
            # Whisper handles offline and online seamlessly and auto-detects languages
            # fp16=False to avoid warnings on CPU
            self.model = whisper.load_model(model_size)
        except Exception as e:
            print(f"[ERROR] Could not load Whisper model. Error: {e}")
            self.model = None

    def listen(self) -> tuple[str, str]:
        """Returns a tuple of (transcribed_text, detected_language_code)"""
        with sr.Microphone() as source:
            self.recognizer.adjust_for_ambient_noise(source, duration=0.2)
            try:
                # Capture the audio input
                audio = self.recognizer.listen(source, timeout=5, phrase_time_limit=15)
            except sr.WaitTimeoutError:
                return ("", "en")
            
            if self.model:
                try:
                    # Whisper expects 16kHz float32 NumPy array
                    wav_data = audio.get_raw_data(convert_rate=16000, convert_width=2)
                    audio_np = np.frombuffer(wav_data, dtype=np.int16).astype(np.float32) / 32768.0
                    
                    # Transcribe and Auto-Detect language!
                    result = self.model.transcribe(audio_np, fp16=False)
                    text = result.get('text', '').strip()
                    lang = result.get('language', 'en')
                    
                    if text:
                        return (text, lang)
                except Exception as e:
                    print(f"[ERROR] Whisper ASR failed: {e}")
                    
        return ("", "en")

import os
import subprocess

def speak(text: str, lang: str = 'en'):
    print(f"JK: {text}")
    try:
        if lang == 'en':
            # Run SAPI in a completely isolated PowerShell process to avoid Python PyAudio COM threading bugs
            clean_text = text.replace('\n', ' ').replace('\r', '').replace('"', '').replace("'", "")
            cmd = f'powershell -Command "Add-Type -AssemblyName System.Speech; $synth = New-Object System.Speech.Synthesis.SpeechSynthesizer; $synth.Speak(\'{clean_text}\')"'
            subprocess.run(cmd, shell=True)
        else:
            # For non-English (Tamil, Telugu, etc.), use Google TTS.
            from gtts import gTTS
            import pygame
            
            audio_file = "temp_reply.mp3"
            tts = gTTS(text=text, lang=lang)
            tts.save(audio_file)
            
            pygame.mixer.init()
            pygame.mixer.music.load(audio_file)
            pygame.mixer.music.play()
            
            # Wait for audio to finish playing
            while pygame.mixer.music.get_busy():
                pygame.time.Clock().tick(10)
                
            pygame.mixer.music.unload()
            pygame.mixer.quit()
            
            if os.path.exists(audio_file):
                os.remove(audio_file)
    except Exception as e:
        print(f"[ERROR] TTS Error: {e}")

from llama_cpp import Llama
import os
import json

class LLMEngine:
    _model = None
    MODEL_PATH = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), "models", "llama-3.2-1b-q4_k_m.gguf")
    
    @classmethod
    def get_model(cls) -> Llama:
        if cls._model is None:
            if not os.path.exists(cls.MODEL_PATH):
                raise FileNotFoundError(f"Model not found at {cls.MODEL_PATH}. Run download script.")
            cls._model = Llama(
                model_path=cls.MODEL_PATH,
                n_ctx=2048,        # Context window
                n_threads=4,       # CPU threads
                n_gpu_layers=20,   # Offload layers to GPU if available
                verbose=False,
            )
        return cls._model
    
    @classmethod
    def enhance_text(cls, rough_text: str, context: str = "email") -> str:
        """Convert rough input to professional text."""
        prompt = f"""You are a professional writer. Convert this rough text to 
professional {context}. Return ONLY the enhanced text, nothing else.

Rough: {rough_text}
Professional:"""
        model = cls.get_model()
        result = model(prompt, max_tokens=256, temperature=0.3, stop=["\n\n"])
        return result['choices'][0]['text'].strip()
    
    @classmethod
    def classify_command(cls, raw: str) -> dict:
        """Parse ambiguous command to structured intent."""
        prompt = f"""Classify this command into JSON with fields:
intent, action, target, value. 
Command: "{raw}"
JSON:"""
        model = cls.get_model()
        result = model(prompt, max_tokens=100, temperature=0.1)
        try:
            return json.loads(result['choices'][0]['text'].strip())
        except:
            return {"intent": "unknown", "action": None, "target": None, "value": None}

    @classmethod
    def chat(cls, text: str, target_lang: str = 'en') -> str:
        """Generate a conversational response like Jarvis."""
        from engine.memory.memory_manager import MemoryManager
        memory = MemoryManager.get_instance()
        context = memory.get_context_string()
        
        prompt = f"You are Jarvis, a helpful and precise AI assistant. Respond directly to the user's input without adding unnecessary details or making up scenarios.\n{context}User: {text}\nJarvis:"
        
        model = cls.get_model()
        result = model(prompt, max_tokens=100, temperature=0.5, stop=["\n", "User:"])
        reply = result['choices'][0]['text'].strip()
        
        # Fallback if the model outputs nothing
        if not reply:
            reply = "I'm not quite sure what you mean, sir."
            
        # Save interaction to memory (in English)
        memory.add_interaction(text, reply)
            
        # Translate reply to the target language if it's not English
        if target_lang != 'en':
            try:
                from googletrans import Translator
                translator = Translator()
                reply = translator.translate(reply, src='en', dest=target_lang).text
            except Exception as e:
                print(f"[ERROR] Could not translate response to {target_lang}: {e}")
                
        return reply

import re
from dataclasses import dataclass
from typing import Optional

@dataclass
class ParsedCommand:
    intent: str
    target: Optional[str]
    action: Optional[str]
    value: Optional[str]
    raw: str

class CommandParser:
    PATTERNS = {
        r'(hi|hello|hey).*':               ('greeting', 'greet', None, None),
        r'unlock phone':                   ('device', 'unlock', None, None),
        r'lock (the )?phone':              ('device', 'lock', None, None),
        r'.*(turn on|turn off|enable|disable) (light|lights|room light).*': ('home', 'toggle_light', r'\2', r'\1'),
        r'.*turn (on|off) (wifi|wi-fi).*': ('device', 'wifi', None, 1),
        r'.*turn (on|off) bluetooth.*':    ('device', 'bluetooth', None, 1),
        r'.*brightness.*(\d+).*':          ('device', 'set_brightness', None, 1),
        r'.*volume.*(\d+).*':              ('device', 'volume', None, 1),
        r'.*battery.*':                    ('device', 'check_battery', None, None),
        r'.*system status.*':              ('device', 'check_system', None, None),
        r'.*(what do you see|what am i holding|look at this|what is this).*': ('vision', 'analyze', None, None),
        r'(open|launch) (.+)':             ('app', 'open', 2, None),
        r'close (.+)':                     ('app', 'close', 1, None),
        r'call (.+)':                      ('comms', 'call', 1, None),
        r'attend (the )?call':             ('comms', 'attend', None, None),
        r'reject (the )?call':             ('comms', 'reject', None, None),
        r'send whatsapp "(.+)" to (.+)':   ('comms', 'whatsapp', 2, 1),
        r'send sms "(.+)" to (.+)':        ('comms', 'sms', 2, 1),
        r'read notifications':             ('comms', 'read_notifs', None, None),
        r'send (mail|email) to (.+) saying (.+)': ('email', 'compose_with_content', 2, 3),
        r'send (mail|email) to (.+)':      ('email', 'compose', 2, None),
        r'reply "(.+)"':                   ('comms', 'reply', None, 1),
    }
    
    def parse(self, raw: str, lang: str = 'en') -> ParsedCommand:
        if not raw:
            return ParsedCommand("unknown", None, None, None, raw)
            
        original_raw = raw
        
        # Translate to English so our hardcoded commands and LLM still work!
        if lang != 'en':
            try:
                from googletrans import Translator
                translator = Translator()
                raw = translator.translate(raw, src=lang, dest='en').text
                print(f"[Translated to English: {raw}]")
            except Exception as e:
                print(f"[Translation Error] {e}")

        raw_lower = raw.lower().strip()
        # Remove "jk" prefix if present
        raw_lower = re.sub(r'^jk\s+', '', raw_lower)
        
        for pattern, (intent, action, target_group, value_group) in self.PATTERNS.items():
            match = re.match(pattern, raw_lower)
            if match:
                return ParsedCommand(
                    intent=intent,
                    action=action,
                    target=match.group(target_group) if target_group else None,
                    value=match.group(value_group) if value_group else None,
                    raw=original_raw
                )
                
        return self._llm_parse(raw, original_raw, lang)

    def _llm_parse(self, raw: str, original_raw: str, lang: str) -> ParsedCommand:
        # Delegate ambiguous commands to local LLM
        from engine.ai.local_llm import LLMEngine
        result = LLMEngine.classify_command(raw)
        
        intent = result.get("intent", "unknown")
        
        # Clamp hallucinated intents from the small 1B model
        valid_intents = {"device", "app", "comms", "email", "home", "greeting", "vision", "chat"}
        if intent not in valid_intents:
            intent = "unknown"
        
        if intent == "unknown":
            # If it's not a recognizable command, treat it as a conversation!
            reply = LLMEngine.chat(raw, target_lang=lang)
            return ParsedCommand(
                intent="chat",
                action="respond",
                target=None,
                value=reply,
                raw=original_raw
            )
            
        return ParsedCommand(
            intent=intent,
            action=result.get("action"),
            target=result.get("target"),
            value=result.get("value"),
            raw=original_raw
        )

def parse_command(text: str, lang: str = 'en') -> dict:
    """Wrapper for backwards compatibility with executor.py"""
    parser = CommandParser()
    parsed = parser.parse(text, lang)
    return {
        "intent": parsed.intent,
        "action": parsed.action,
        "target": parsed.target,
        "value": parsed.value,
        "lang": lang
    }

import os
import json

class MemoryManager:
    _instance = None
    
    @classmethod
    def get_instance(cls):
        if cls._instance is None:
            cls._instance = MemoryManager()
        return cls._instance

    def __init__(self, history_file="memory.json", max_history=10):
        self.history_file = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), history_file)
        self.max_history = max_history
        self.history = self._load_history()

    def _load_history(self):
        if os.path.exists(self.history_file):
            try:
                with open(self.history_file, 'r', encoding='utf-8') as f:
                    return json.load(f)
            except Exception:
                return []
        return []

    def _save_history(self):
        try:
            with open(self.history_file, 'w', encoding='utf-8') as f:
                json.dump(self.history, f, indent=4)
        except Exception as e:
            print(f"[ERROR] Could not save memory: {e}")

    def add_interaction(self, user_text, jarvis_text):
        if not user_text.strip() or not jarvis_text.strip():
            return
            
        self.history.append({"user": user_text, "jarvis": jarvis_text})
        if len(self.history) > self.max_history:
            self.history.pop(0)
        self._save_history()

    def get_context_string(self):
        if not self.history:
            return ""
        context = "Context:\n"
        # Only inject the last 3 interactions to avoid overflowing the LLM context window
        for interaction in self.history[-3:]: 
            context += f"User: {interaction['user']}\nJarvis: {interaction['jarvis']}\n"
        context += "\n"
        return context

    def clear(self):
        self.history = []
        self._save_history()

import psutil
import screen_brightness_control as sbc

class DeviceController:
    @staticmethod
    def get_battery_status() -> str:
        try:
            battery = psutil.sensors_battery()
            if battery:
                percent = battery.percent
                plugged = battery.power_plugged
                status = "charging" if plugged else "discharging"
                return f"Battery is at {percent} percent and is currently {status}."
            return "I am unable to read the battery status on this device."
        except Exception as e:
            return f"Error reading battery: {e}"
            
    @staticmethod
    def get_system_stats() -> str:
        try:
            cpu = psutil.cpu_percent(interval=0.5)
            ram = psutil.virtual_memory().percent
            return f"CPU usage is at {cpu} percent, and Memory usage is at {ram} percent."
        except Exception as e:
            return f"Error reading system stats: {e}"
        
    @staticmethod
    def set_brightness(level: int) -> str:
        try:
            # clamp between 0 and 100
            level = max(0, min(100, level))
            sbc.set_brightness(level)
            return f"Screen brightness set to {level} percent."
        except Exception as e:
            return f"I couldn't adjust the brightness. The monitor might not support DDC/CI control via script."

    @staticmethod
    def toggle_wifi(state: str) -> str:
        # Mocking for safety. Toggling real Wi-Fi on Windows requires admin privileges via netsh.
        if state.lower() == 'on':
            return "Wi-Fi interface has been enabled."
        else:
            return "Wi-Fi interface has been disabled."

import cv2
import os
import numpy as np

class VisionEngine:
    def __init__(self):
        self.models_dir = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), "models", "vision")
        os.makedirs(self.models_dir, exist_ok=True)
        
        self.prototxt = os.path.join(self.models_dir, "MobileNetSSD_deploy.prototxt.txt")
        self.model_path = os.path.join(self.models_dir, "MobileNetSSD_deploy.caffemodel")
        
        # Pascal VOC classes
        self.CLASSES = ["background", "aeroplane", "bicycle", "bird", "boat",
            "bottle", "bus", "car", "cat", "chair", "cow", "diningtable",
            "dog", "horse", "motorbike", "person", "pottedplant", "sheep",
            "sofa", "train", "tvmonitor", "cell phone"]

        self.net = None
        if os.path.exists(self.prototxt) and os.path.exists(self.model_path):
            try:
                self.net = cv2.dnn.readNetFromCaffe(self.prototxt, self.model_path)
            except Exception as e:
                print(f"[ERROR] Could not load vision model: {e}")

    def capture_and_analyze(self) -> str:
        if self.net is None:
            return "My vision module is inactive. Please download the MobileNet SSD model files to the models slash vision folder."

        try:
            # Open webcam
            cap = cv2.VideoCapture(0)
            if not cap.isOpened():
                return "I am unable to access the camera hardware."
                
            ret, frame = cap.read()
            cap.release()
            
            if not ret:
                return "I couldn't capture an image from the camera."
                
            # Analyze frame
            (h, w) = frame.shape[:2]
            blob = cv2.dnn.blobFromImage(cv2.resize(frame, (300, 300)), 0.007843, (300, 300), 127.5)
            self.net.setInput(blob)
            detections = self.net.forward()
            
            found_objects = set()
            for i in np.arange(0, detections.shape[2]):
                confidence = detections[0, 0, i, 2]
                if confidence > 0.5:
                    idx = int(detections[0, 0, i, 1])
                    if idx < len(self.CLASSES):
                        found_objects.add(self.CLASSES[idx])
                        
            if "person" in found_objects:
                found_objects.remove("person") # ignore the user
                
            if not found_objects:
                return "I don't see any distinct objects."
                
            objects_str = ", ".join(list(found_objects))
            return f"It looks like you are holding or showing me a {objects_str}."
        except Exception as e:
            return f"An error occurred while trying to see: {e}"

def download_file(url, filepath):
    if os.path.exists(filepath):
        print(f"[OK] {os.path.basename(filepath)} already exists.")
        return

    os.makedirs(os.path.dirname(filepath), exist_ok=True)
    print(f"[FETCH] Downloading {os.path.basename(filepath)}...")
    try:
        with urllib.request.urlopen(url, context=context) as response, open(filepath, 'wb') as out_file:
            data = response.read()
            out_file.write(data)
        print(f"[DONE] Downloaded {os.path.basename(filepath)} successfully.")
    except Exception as e:
        print(f"[ERROR] Failed to download {os.path.basename(filepath)}: {e}")

def progress_bar(count, block_size, total_size):
    if total_size == -1: return
    percent = int(count * block_size * 100 / total_size)
    sys.stdout.write(f"\rDownloading... {percent}%")
    sys.stdout.flush()

if __name__ == "__main__":
    os.makedirs(MODELS_DIR, exist_ok=True)
    for filename, url in MODELS.items():
        filepath = os.path.join(MODELS_DIR, filename)
        download_file(url, filepath)

class EmailWriter:
    SYSTEM_PROMPT = """You are a professional email assistant. 
Convert rough input into polished, professional communication.
Keep it concise, warm, and appropriate. Return ONLY the email body."""
    
    def compose_from_rough(self, rough: str, recipient: str = "Sir/Ma'am") -> str:
        """
        Input:  "send mail to sir i cant attend class today"
        Output: "Dear Sir,\n\nI hope you are doing well. I would like 
                 to inform you that I will not be able to attend 
                 today's class. Thank you for your understanding.\n\nRegards"
        """
        from engine.ai.local_llm import LLMEngine
        
        # Extract just the message content
        rough_message = rough.lower().replace("send mail to", "").replace("send email to", "").strip()
        # Remove recipient prefix  
        parts = rough_message.split(' ', 1)
        if len(parts) > 1:
            rough_message = parts[1]
        
        enhanced = LLMEngine.enhance_text(
            rough_text=rough_message,
            context=f"professional email to {recipient}"
        )
        return f"Dear {recipient.title()},\n\n{enhanced}\n\nRegards"
    
    def correct_grammar(self, text: str) -> str:
        from engine.ai.local_llm import LLMEngine
        return LLMEngine.enhance_text(text, context="grammatically correct professional message")
